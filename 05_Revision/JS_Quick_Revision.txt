[ JAVASCRIPT QUICK REVISION CHEAT SHEET ]

1. VARIABLES
   const x = 10;            -> Cannot change (Use by default)
   let y = 10;              -> Can change
   var z = 10;              -> Old (Avoid)

2. DATA TYPES
   String ("Text"), Number (123), Boolean (true/false),
   Null (Empty), Undefined (No value), Object ({}), Array ([])

3. FUNCTIONS
   // Arrow Function
   const add = (a, b) => a + b;

   // Standard
   function add(a, b) { return a + b; }

4. ARRAYS
   const arr = [1, 2, 3];
   arr.push(4);             -> Add to end
   arr.pop();               -> Remove from end
   arr.map(x => x * 2);     -> Transform all items
   arr.filter(x => x > 1);  -> Select specific items

5. OBJECTS
   const person = { name: "John", age: 30 };
   console.log(person.name);
   const { name } = person; -> Destructuring

6. DOM MANIPULATION
   document.querySelector(".class"); -> Select Element
   element.innerText = "Text";       -> Change Text
   element.style.color = "red";      -> Change Style
   element.classList.add("active");  -> Add Class

7. EVENTS
   btn.addEventListener("click", (e) => {
       console.log("Clicked!");
       e.preventDefault(); // Stop form reload
   });

8. ASYNC
   // Modern Async/Await
   async function getData() {
       try {
           let res = await fetch("url");
           let data = await res.json();
       } catch (err) {
           console.error(err);
       }
   }
